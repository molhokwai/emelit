import markdown

def extract_sections(md_file):
  """Extracts the sections from a Markdown file.

  Args:
    md_file: The path to the Markdown file.

  Returns:
    A list of sections, where each section is a string.
  """

  with open(md_file, 'r') as f:
    markdown_text = f.read()

  markdown_document = markdown.markdown(markdown_text)

  sections = []
  for section in markdown_document.split('\n---'):
    if section:
      sections.append(section)

  return sections

if __name__ == '__main__':
  md_file = 'my_file.md'

  sections = extract_sections(md_file)

  for section in sections:
    print(section)

That being said, I'm a Lover of the distribution since the 1st time I saw the system running cold under 400MB of memory consumption, and after ~2 days of frustration, I'm back to being a faithful Lover & User of Lubuntu...
